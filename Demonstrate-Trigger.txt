CREATE TABLE STUDENTS (
    STUDENT_ID NUMBER PRIMARY KEY,
    STUDENT_NAME VARCHAR2(50)
);

CREATE OR REPLACE TRIGGER BEFORE_INSERT_STUDENT
BEFORE INSERT ON STUDENTS
FOR EACH ROW
BEGIN
    
    :NEW.STUDENT_NAME := UPPER(:NEW.STUDENT_NAME);
END;
