-- Create EMPLOYEES_A table
CREATE TABLE EMPLOYEES_A (
    EMP_ID NUMBER PRIMARY KEY,
    EMP_NAME VARCHAR2(50),
    DEPT_ID NUMBER
);

-- Create EMPLOYEES_B table
CREATE TABLE EMPLOYEES_B (
    EMP_ID NUMBER PRIMARY KEY,
    EMP_NAME VARCHAR2(50),
    DEPT_ID NUMBER
);

-- Insert sample data into EMPLOYEES_A
INSERT INTO EMPLOYEES_A (EMP_ID, EMP_NAME, DEPT_ID) VALUES (1, 'John Doe', 101);
INSERT INTO EMPLOYEES_A (EMP_ID, EMP_NAME, DEPT_ID) VALUES (2, 'Jane Smith', 102);
INSERT INTO EMPLOYEES_A (EMP_ID, EMP_NAME, DEPT_ID) VALUES (3, 'Sam Brown', 103);

-- Insert sample data into EMPLOYEES_B
INSERT INTO EMPLOYEES_B (EMP_ID, EMP_NAME, DEPT_ID) VALUES (2, 'Jane Smith', 102);
INSERT INTO EMPLOYEES_B (EMP_ID, EMP_NAME, DEPT_ID) VALUES (4, 'Alice White', 104);
INSERT INTO EMPLOYEES_B (EMP_ID, EMP_NAME, DEPT_ID) VALUES (5, 'Bob Johnson', 105);

COMMIT;

--union
SELECT EMP_ID, EMP_NAME, DEPT_ID
FROM EMPLOYEES_A
UNION
SELECT EMP_ID, EMP_NAME, DEPT_ID
FROM EMPLOYEES_B;

--intersect
SELECT EMP_ID, EMP_NAME, DEPT_ID
FROM EMPLOYEES_A
INTERSECT
SELECT EMP_ID, EMP_NAME, DEPT_ID
FROM EMPLOYEES_B;

--minus
SELECT EMP_ID, EMP_NAME, DEPT_ID
FROM EMPLOYEES_A
MINUS
SELECT EMP_ID, EMP_NAME, DEPT_ID
FROM EMPLOYEES_B;
